  setopt noshwordsplit

  autoload -Uz colors && colors
  local B=$'\e[1m' Dim=$'\e[2m' R=$'\e[0m'
  local C="${fg[cyan]}" Y="${fg[yellow]}" G="${fg[green]}" reset="${reset_color}"

  local -a FUNCS=(

}
